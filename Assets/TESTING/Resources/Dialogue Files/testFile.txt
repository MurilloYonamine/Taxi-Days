$explorationPoints = 0
$combatPoints = 0
$diplomacyPoints = 0
$mysticalPoints = 0

CreateCharacter(Lissima), CreateCharacter(Cind), CreateCharacter(Gato)
Lissima.Show()
Cind.Show()
Gato.Show()

Lissima.Move(0.2 0.5)
Cind.Move(0.8 0.5)
Gato.Move(0.45 0.5)

Lissima "Você já pensou em qual tipo de aventura seria mais emocionante para nós?"

Cind "Eu adoro a ideia de explorar novos lugares. Sempre há algo novo para descobrir!"

Gato "Exploração é ótimo, mas e se a gente se deparar com um desafio inesperado?"

Lissima "É verdade! Mas isso só torna as coisas mais emocionantes, não é?"

Cind "Exatamente! Mas também gosto de um bom combate para testar minhas habilidades."

Gato "Eu prefiro evitar conflitos sempre que possível. Diplomacia é o caminho!"

Lissima "Cada um tem suas preferências, mas podemos decidir juntos."

choice "Qual tipo de aventura você prefere?"
{
    - Exploração
        Lissima "Explorar novos lugares é incrível! Eu amo descobrir segredos escondidos."
        mainChar "Você já encontrou algo fascinante em suas explorações?"
        Lissima "Uma vez encontrei uma caverna cheia de cristais brilhantes! {c} Foi mágico!"
        $explorationPoints += 10
        narrator "Pontos de Exploração: $explorationPoints"
    - Combate
        Cind "Nada como uma boa batalha para testar suas habilidades!"
        mainChar "Você gosta de lutas, Cind?"
        Cind "Sim! {a} Embora eu prefira uma luta justa. Sempre tenho um plano!"
        "Mas o importante é o que você aprende com a experiência."
        $combatPoints += 10
        narrator "Pontos de Combate: $combatPoints"
    - Diplomacia
        Gato "Eu prefiro resolver conflitos com palavras, não com espadas."
        mainChar "Você acha que sempre funciona?"
        Gato "Nem sempre, mas é a melhor maneira de evitar derramamento de sangue."
        $diplomacyPoints += 10
        narrator "Pontos de Diplomacia: $diplomacyPoints"
        choice "E se as palavras falharem?" 
        {
            - "Devemos estar preparados!"
                Gato "Exato! Um bom diálogo pode salvar vidas."
            - "Mas algumas situações exigem ação."
                Gato "Concordo! Às vezes, uma combinação de palavras e ações é a chave."
        }
    - Aventura Mística
        Lissima "Eu amo as histórias de magia e mistério!"
        mainChar "Você acredita que as criaturas mágicas realmente existem?"
        Lissima "Com certeza! Eu gostaria de encontrar um dragão um dia!"
        Gato "Cuidado! Dragões podem ser traiçoeiros. {a} Mas são fascinantes!"
        "Imagina a quantidade de conhecimento que um dragão pode compartilhar!"
        $mysticalPoints += 10
        narrator "Pontos de Aventura Mística: $mysticalPoints"
}
